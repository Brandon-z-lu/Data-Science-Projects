library(ggplot2)

generate_trend <- function(start, mean, sd, typeOfEnergy, n = 27) {
  values <- rnorm(n, mean = mean, sd = sd)

  trend <- cumsum(values)
  trend <- pmax(trend + start, 0)

  df <- data.frame(
    year = 2023:(2023 + n - 1),
    capacity = trend,
    typeOfEnergy = rep(typeOfEnergy, n)
  )

  df$year <- paste(df$year, df$year + 1, sep = "-")

  return(df)
}
